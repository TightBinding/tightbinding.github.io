159  void resize(std::array<size_t,_dim> new_sizes) {
160  // copy new size
161  _sizes = new_sizes;
162 
163  // Strides
164  // Example for _dim=5, let n = _sizes
165  // _strides[0] = n[1]*n[2]*n[3]*n[4];
166  // _strides[1] = n[2]*n[3]*n[4];
167  // _strides[2] = n[3]*n[4];
168  // _strides[3] = n[4];
169  // _strides[4] = 1;
170  //
171  // Loop below generalizes above for arbitrary dimension using:
172  // _strides[i] = _size[data_dim-1]*_size[data_dim-2]..._size[i+1]
173  for(size_t i=0; i < _dim; i++) {
174  _strides[i] = 1;
175  for(size_t j=_dim-1; j > i; j--) {
176  _strides[i] *= _sizes[j];
177  }
178  }
179 
180  // Total number of elements
181  size_t elem = 1;
182  for(size_t i=0; i<_dim; i++)
183  elem *= _sizes[i];
184 
185  // Allocate new data
186  _data.resize(elem);
187  }

242  template<typename... Index>
243  inline T& operator()(Index... index) {
244  static_assert(sizeof...(index) == _dim, "Invalid dimension for index");
245 
246  // copy into array
247  size_t a_index[_dim]{size_t(index)...};
248 
249 #ifdef TBPP_NARRAY_RANGE_CHECK
250  for(size_t i=0; i<_dim; i++)
251  if(a_index[i] >= _sizes[i])
252  throw std::out_of_range("Index out of range");
253 #endif
254 
255  size_t raw_index = std::inner_product(a_index, a_index+N,
256  _strides.begin(), size_t(0));
257  return _data[raw_index];
258  }

349  std::ofstream out(filename, std::ios::out | std::ios::trunc);

355  } else {
356  out.precision(std::numeric_limits<double>::max_digits10);
357  }
358  // Set output format
359  out.setf(std::ios::scientific);
360 
361  // header
362  out << "NArray sparse\n"
363  << "dim " << _dim << '\n';
364  out << "sizes";
365  for(size_t i=0; i<_dim; i++) {
366  out << ' ' << _sizes[i];
367  }
368  out << '\n';
369 
370  // data
371  for(size_t i=0; i< elem(); i++) {
372  if(_data[i] != 0) {
373  size_t index = i;
374  for(size_t j=0; j<_dim; j++) {
375  size_t id = index/_strides[j];
376  index -= id*_strides[j];
377  out << id << ' ';
378  }
379  out << _data[i] << '\n';
380  }
381  if(!out.good()) throw std::runtime_error("error writing to file");
382  }
